

mkdir -p ~/.ssh
chmod 700 ~/.ssh

# Generate the guest's own keypair. build.sh exports ~/.ssh/id_rsa.pub as
# <output>-id_rsa.pub (the key the VM uses to connect back to the build host
# for rsync/sshfs), so it must exist before build.sh reads it.
rm -f ~/.ssh/id_rsa
ssh-keygen -t rsa -f ~/.ssh/id_rsa -q -N ''

# Do NOT reconfigure or restart sshd here. Root key login is already set up by
# hooks/prepareImage.sh (PermitRootLogin / PubkeyAuthentication / AcceptEnv are
# appended to /etc/ssh/sshd_config before first boot). Ubuntu/Debian openssh uses
# socket activation, and "systemctl restart ssh" can drop the listener -- under
# aarch64 TCG emulation that left sshd unreachable and hung the build.
